1//! Types and operations specific to handling of transparent addresses and transparent outputs
2//! received by the wallet.
3use transparent::keys::TransparentKeyScope;

5/// Configuration for gap limits used in transparent address management.
6///
7/// A "gap limit" is the number of consecutive unused addresses that the wallet will generate
8/// before stopping. This concept comes from BIP-44 HD wallet standards: when scanning for
9/// funds, the wallet generates addresses sequentially and stops when it encounters a "gap"
10/// of unused addresses equal to this limit. In recovery, a wallet will discover funds
11/// belonging to generated addresses. Note that order of operations matters; if a wallet scans
12/// blocks for transparent outputs instead of querying the UTXO set, out-of-order scanning may
13/// result in funds not being found; as a result, wallets should still fall back to making
14/// temporally unlinkable, private queries to the UTXO set for each address controlled by the
15/// wallet at any time that a gap in scanned blocks is introduced by out-of-order scanning.
16///
17/// In Zcash, we define individual gap limits for the following address types:
18/// - **External addresses**: Addresses shared with external parties.
19/// - **Internal (change) addresses**: Used for transparent change outputs.
20/// - **Ephemeral addresses**: Used for single-use purposes like ZIP-320 TEX address transfers.
21#[derive(Debug, Copy, Clone, PartialEq, Eq)]
22pub struct GapLimits {
23    external: u32,
24    internal: u32,
25    ephemeral: u32,
26}
27
28impl GapLimits {
29    /// Constructs a new `GapLimits` value from its constituent parts.
30    ///
31    /// The gap limits recommended for use with this crate are supplied by the [`Default`]
32    /// implementation for this type.
33    ///
34    /// This constructor is only available under the `unstable` feature, as it is not recommended
35    /// for general use.
36    #[cfg(any(test, feature = "test-dependencies", feature = "unstable"))]
37    pub fn new(external: u32, internal: u32, ephemeral: u32) -> Self {
38        Self {
39            external,
40            internal,
41            ephemeral,
42        }
43    }
44
45    /// Returns the gap limit for external-scoped addresses.
46    pub fn external(&self) -> u32 {
47        self.external
48    }
49
50    /// Returns the gap limit for internal-scoped addresses.
51    pub fn internal(&self) -> u32 {
52        self.internal
53    }
54
55    /// Returns the gap limit for ephemeral-scoped addresses.
56    pub fn ephemeral(&self) -> u32 {
57        self.ephemeral
58    }
59
60    /// Returns the gap limit for the given transparent key scope, or `None` if the key scope is
61    /// one for which gap limits are not managed by this type.
62    pub fn limit_for(&self, scope: TransparentKeyScope) -> Option<u32> {
63        match scope {
64            TransparentKeyScope::EXTERNAL => Some(self.external()),
65            TransparentKeyScope::INTERNAL => Some(self.internal()),
66            TransparentKeyScope::EPHEMERAL => Some(self.ephemeral()),
67            _ => None,
68        }
69    }
70}
71
72/// The default gap limits supported by this implementation are:
73///
74/// - external addresses: 10
75/// - transparent internal (change) addresses: 5
76/// - ephemeral addresses: 10
77///
78/// These limits are chosen with the following rationale:
79/// - At present, many wallets query light wallet servers with a set of addresses, because querying
80///   for each address independently and in a fashion that is not susceptible to clustering via
81///   timing correlation leads to undesirable delays in discovery of received funds. As such, it is
82///   desirable to minimize the number of addresses that can be "linked", i.e. understood by the
83///   light wallet server to all belong to the same wallet.
84/// - For transparent change addresses it is always expected that an address will receive funds
85///   immediately following its generation except in the case of wallet failure.
86/// - For externally-scoped transparent addresses and ephemeral addresses, it is desirable to use a
87///   slightly larger gap limit to account for addresses that were shared with counterparties never
88///   having been used. However, we don't want to use the full 20-address gap limit space because
89///   it's possible that in the future, changes to the light wallet protocol will obviate the need to
90///   query for UTXOs in a fashion that links those addresses to one another. In such a
91///   circumstance, the gap limit will be adjusted upward and address rotation should then choose
92///   an address that is outside the current gap limit; after that change, newly generated
93///   addresses will not be exposed as linked in the view of the light wallet server.
94impl Default for GapLimits {
95    fn default() -> Self {
96        Self {
97            external: 10,
98            internal: 5,
99            ephemeral: 10,
100        }
101    }
